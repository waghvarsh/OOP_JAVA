class Calculator {
    public static void main(String[] args) {
        int i=5;
        double d=10.00;
        long l=15;
        float f =15.00f;

        double dResult = i+d;
        long lResult = i+l;
        float fResult= i+f;

        System.out.println("double : "+dResult);
        System.out.println("long : "+lResult);
        System.out.println("float : "+fResult);
    }
}
